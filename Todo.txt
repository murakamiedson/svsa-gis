SELECT
	endereco.endereco,
    endereco.numero,
    endereco.bairro,
    pessoa.sexo,
    situacaoviolencia.situacao,
    unidade.codigo,
    unidade.nome
FROM svsaweb.pessoa
INNER JOIN familia ON pessoa.codigo_familia = familia.codigo
INNER JOIN endereco ON familia.codigo_endereco = endereco.codigo
INNER JOIN prontuario ON familia.codigo = prontuario.codigo_familia
INNER JOIN unidade ON prontuario.codigo_unidade = unidade.codigo
LEFT JOIN situacaoviolencia ON pessoa.codigo = situacaoviolencia.codigo_pessoa;
